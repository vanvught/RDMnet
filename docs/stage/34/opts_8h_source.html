288 /*
289  * [DEPRECATED]
290  * \brief Whether to allow sockets associated with connections to be polled externally.
291  *
292  * If this option is defined to 0, the externally-managed-socket functions in the RDMnet connection
293  * API will not be implemented. Most applications will want the default behavior, unless scaling
294  * the number of connections is a concern. Broker applications will set this to 1.
295  */
296 #ifndef RDMNET_ALLOW_EXTERNALLY_MANAGED_SOCKETS
297 #define RDMNET_ALLOW_EXTERNALLY_MANAGED_SOCKETS RDMNET_FULL_OS_AVAILABLE_HINT
298 #endif
299 
301 /*
302  * The library has some limitations around static memory allocation and how many message structures
303  * can be allocated at a time. If #RDMNET_ALLOW_EXTERNALLY_MANAGED_SOCKETS is true, the library has
304  * no guarantee as to how many threads could be receiving and allocating messages simultaneously;
305  * therefore, in this case, #RDMNET_DYNAMIC_MEM must be enabled.
306  */
307 #if (RDMNET_ALLOW_EXTERNALLY_MANAGED_SOCKETS && !RDMNET_DYNAMIC_MEM)
308 #error "RDMNET_ALLOW_EXTERNALLY_MANAGED_SOCKETS=1 requires RDMNET_DYNAMIC_MEM=1"
309 #endif
