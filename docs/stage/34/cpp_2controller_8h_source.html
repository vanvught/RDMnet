426 inline etcpal::Error Controller::Startup(NotifyHandler& notify_handler, const Settings& settings,
427  const RdmData& rdm_data)
428 {
429  if (!settings.IsValid() || !rdm_data.IsValid())
430  return kEtcPalErrInvalid;
431 
432  notify_ = &notify_handler;
433  my_rdm_data_ = rdm_data;
434 
435  // clang-format off
436  RdmnetControllerConfig config = {
437  settings.cid.get(), // CID
438  { // Callback shims
439  internal::ControllerLibCbConnected,
440  internal::ControllerLibCbConnectFailed,
441  internal::ControllerLibCbDisconnected,
442  internal::ControllerLibCbClientListUpdate,
443  internal::ControllerLibCbRdmResponseReceived,
444  internal::ControllerLibCbStatusReceived,
445  internal::ControllerLibCbResponderIdsReceived,
446  &notify_handler // Context
447  },
448  { // RDM command callback shims
449  nullptr, nullptr, nullptr, nullptr
450  },
451  { // RDM data
452  rdm_data.manufacturer_label.c_str(),
453  rdm_data.device_model_description.c_str(),
454  rdm_data.software_version_label.c_str(),
455  rdm_data.device_label.c_str(),
456  rdm_data.device_label_settable
457  },
458  settings.uid.get(), // UID
459  settings.search_domain.c_str(), // Search domain
460  settings.create_llrp_target, // Create LLRP target
461  nullptr,
462  0
463  };
464  // clang-format on
465 
466  // LLRP network interfaces
467  if (!settings.llrp_netints.empty())
468  {
469  config.llrp_netints = settings.llrp_netints.data();
470  config.num_llrp_netints = settings.llrp_netints.size();
471  }
472 
473  return rdmnet_controller_create(&config, &handle_);
474 }
