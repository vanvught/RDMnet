421 inline etcpal::Error Controller::Startup(ControllerNotifyHandler& notify_handler, const ControllerSettings& settings,
422  const ControllerRdmData& rdm_data)
423 {
424  if (!settings.IsValid() || !rdm_data.IsValid())
425  return kEtcPalErrInvalid;
426 
427  notify_ = &notify_handler;
428  my_rdm_data_ = rdm_data;
429 
430  // clang-format off
431  RdmnetControllerConfig config = {
432  settings.cid.get(), // CID
433  { // Callback shims
434  internal::ControllerLibCbConnected,
435  internal::ControllerLibCbConnectFailed,
436  internal::ControllerLibCbDisconnected,
437  internal::ControllerLibCbClientListUpdate,
438  internal::ControllerLibCbRdmResponseReceived,
439  internal::ControllerLibCbStatusReceived,
440  internal::ControllerLibCbResponderIdsReceived
441  },
442  { // RDM command callback shims
443  nullptr, nullptr, nullptr
444  },
445  { // RDM data
446  rdm_data.manufacturer_label.c_str(),
447  rdm_data.device_model_description.c_str(),
448  rdm_data.software_version_label.c_str(),
449  rdm_data.device_label.c_str(),
450  rdm_data.device_label_settable
451  },
452  &notify_handler, // Callback context
453  settings.uid.get(), // UID
454  settings.search_domain.c_str(), // Search domain
455  settings.create_llrp_target, // Create LLRP target
456  nullptr,
457  0
458  };
459  // clang-format on
460 
461  // LLRP network interfaces
462  if (!settings.llrp_netints.empty())
463  {
464  config.llrp_netints = settings.llrp_netints.data();
465  config.num_llrp_netints = settings.llrp_netints.size();
466  }
467 
468  return rdmnet_controller_create(&config, &handle_);
469 }
