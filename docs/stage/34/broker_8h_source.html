138 inline bool BrokerSettings::IsValid() const
139 {
140  // clang-format off
141  return (
142  !cid.IsNull() &&
143  (!scope.empty() && scope.length() < (E133_SCOPE_STRING_PADDED_LENGTH - 1)) &&
144  (!dns.manufacturer.empty() && dns.manufacturer.length() < (E133_MANUFACTURER_STRING_PADDED_LENGTH - 1)) &&
145  (!dns.model.empty() && dns.model.length() < (E133_MODEL_STRING_PADDED_LENGTH - 1)) &&
146  (!dns.service_instance_name.empty() && dns.service_instance_name.length() < (E133_SERVICE_NAME_STRING_PADDED_LENGTH - 1)) &&
147  (listen_port == 0 || listen_port >= 1024) &&
148  (uid.manufacturer_id() != 0) &&
149  (uid.IsStatic() || uid.IsDynamicUidRequest())
150  );
151  // clang-format on
152 }
