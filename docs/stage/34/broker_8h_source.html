194 inline bool Broker::Settings::IsValid() const
195 {
196  // clang-format off
197  return (
198  !cid.IsNull() &&
199  (!scope.empty() && scope.length() < (E133_SCOPE_STRING_PADDED_LENGTH - 1)) &&
200  (!dns.manufacturer.empty() && dns.manufacturer.length() < (E133_MANUFACTURER_STRING_PADDED_LENGTH - 1)) &&
201  (!dns.model.empty() && dns.model.length() < (E133_MODEL_STRING_PADDED_LENGTH - 1)) &&
202  (!dns.service_instance_name.empty() && dns.service_instance_name.length() < (E133_SERVICE_NAME_STRING_PADDED_LENGTH - 1)) &&
203  (listen_port == 0 || listen_port >= 1024) &&
204  (uid.manufacturer_id() != 0) &&
205  (uid.IsStatic() || uid.IsDynamicUidRequest())
206  );
207  // clang-format on
208 }
